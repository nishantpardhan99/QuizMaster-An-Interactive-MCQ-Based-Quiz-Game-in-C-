// ---------- FUNCTION ----------
int playRound(Question q[], int start) {
    int score = 0, ans;

    for(int i = start; i < start+10; i++) {
        cout << "\n" << q[i].question << endl;
        for(int j = 0; j < 4; j++) {
            cout << j+1 << ". " << q[i].options[j] << endl;
        }

        cout << "Enter answer: ";
        cin >> ans;

        if(ans == q[i].correct) {
            cout << "Correct!\n";
            score++;
        } else {
            cout << "Wrong!\n";
        }
    }
    return score;
}

// ---------- MAIN ----------
int main() {
    int choice;

    while(true) {
        cout << "\n===== QuizMaster =====\n";
        cout << "1. Computer\n2. Food\n3. Sports\n4. Exit\n";
        cout << "Choose category: ";
        cin >> choice;

        if(choice == 4) break;

        Question* selected;
        if(choice == 1) selected = computer;
        else if(choice == 2) selected = food;
        else if(choice == 3) selected = sports;
        else continue;

        int total = 0;

        for(int r = 0; r < 3; r++) {
            cout << "\n--- Round " << r+1 << " ---\n";
            int s = playRound(selected, r*10);
            total += s;

            cout << "Round Score: " << s << "/10\n";
            cout << "Total Score: " << total << "/" << (r+1)*10 << "\n";

            char ch;
            cout << "Press c to continue or s to switch: ";
            cin >> ch;
            if(ch == 's') break;
        }
    }

    cout << "\nThank you for playing!\n";
}
